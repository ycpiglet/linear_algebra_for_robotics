리뷰 빌드의 [수정 제안]이 만든 이슈와 자동 반영 배치의 현황을 GitHub을 열지 않고 한눈에 봅니다.
브라우저에서 GitHub API를 직접 조회하며, 서버는 없습니다.

::: {.callout-note collapse="true"}
## 접근 토큰 (비공개 저장소이거나 조회 한도에 걸릴 때)

저장소가 비공개라면 fine-grained 토큰(권한: Contents·Issues·Pull requests 읽기)을 한 번 입력해 두세요.
이 브라우저의 localStorage에만 저장됩니다.

<p class="text-muted" style="font-size:0.85rem">
편집 이벤트 원장은 저장소의 <code>platform/editorial/events/</code>에 커밋과 함께 쌓입니다.
규약: 배치 PR은 merge commit으로 병합(squash 금지), 특정 교정만 기각하려면 그 커밋을 revert로 상쇄 후 병합.
</p>

```{=html}
<script>
(function editorialDesk() {
  'use strict';
  const REPO = 'ycpiglet/linear_algebra_for_robotics';
  const API = 'https://api.github.com';
  const TOKEN_KEY = 'editorial-desk-token';

  const $ = (id) => document.getElementById(id);

  async function gh(path) {
    const headers = { Accept: 'application/vnd.github+json' };
    const token = localStorage.getItem(TOKEN_KEY);
    if (token) headers.Authorization = 'Bearer ' + token;
    const response = await fetch(API + path, { headers });
    if (!response.ok) {
      throw new Error('GitHub API ' + response.status
        + (response.status === 404 || response.status === 401
           ? ' — 비공개 저장소라면 위의 접근 토큰을 설정하세요.'
           : response.status === 403 ? ' — 조회 한도 초과일 수 있습니다(토큰 설정으로 완화).' : ''));
    }
    return response.json();
  }

  const escapeHtml = (s) => String(s).replace(/[&<>"]/g,
    (c) => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;' }[c]));
  const dateOf = (iso) => iso ? iso.slice(0, 10) : '';

  function issueRow(issue) {
    const labels = issue.labels.map((l) => l.name);
    const state = labels.includes('bridged')
      ? '<span class="badge bg-secondary">브리지 처리됨</span>'
      : '<span class="badge bg-warning text-dark">미처리</span>';
    return '<tr><td><a href="' + issue.html_url + '" target="_blank" rel="noopener">#'
      + issue.number + '</a></td><td>' + escapeHtml(issue.title) + '</td><td>'
      + escapeHtml(issue.user.login) + '</td><td>' + dateOf(issue.created_at)
      + '</td><td>' + state + '</td></tr>';
  }

  // 이슈가 많아져도 총계가 틀리지 않게 최대 3페이지(300건)까지 수거한다.
  async function ghPaged(path, maxPages = 3) {
    const all = [];
    for (let pageNo = 1; pageNo <= maxPages; pageNo += 1) {
      const batch = await gh(path + '&per_page=100&page=' + pageNo);
      all.push(...batch);
      if (batch.length < 100) return { items: all, capped: false };
    }
    return { items: all, capped: true };
  }

  const PAGE_SIZE = 30;

  async function renderPending() {
    const { items, capped } = await ghPaged('/repos/' + REPO + '/issues?labels=editorial&state=open');
    const issues = items.filter((issue) => !issue.pull_request);
    if (!issues.length) { $('desk-pending').innerHTML = '<p>열린 제안이 없습니다. ✨</p>'; return; }
    const fresh = issues.filter((i) => !i.labels.some((l) => l.name === 'bridged')).length;
    const container = $('desk-pending');
    container.innerHTML =
      '<p>' + issues.length + (capped ? '+' : '') + '건 열림 (다음 다이제스트 대상 '
      + fresh + '건 — 매일 06:00 KST 수거)</p>'
      + '<div style="overflow-x:auto"><table class="table table-sm"><thead><tr>'
      + '<th>이슈</th><th>제목</th><th>제안자</th><th>날짜</th><th>상태</th></tr></thead>'
      + '<tbody id="desk-pending-rows"></tbody></table></div>';
    const tbody = document.getElementById('desk-pending-rows');
    const more = document.createElement('button');
    more.type = 'button';
    more.className = 'btn btn-sm btn-outline-secondary';
    let shown = 0;
    const showMore = () => {
      tbody.insertAdjacentHTML('beforeend',
        issues.slice(shown, shown + PAGE_SIZE).map(issueRow).join(''));
      shown = Math.min(shown + PAGE_SIZE, issues.length);
      if (shown >= issues.length) { more.remove(); return; }
      more.textContent = '더 보기 (' + (issues.length - shown) + '건 남음)';
    };
    more.addEventListener('click', showMore);
    container.appendChild(more);
    showMore();
  }

  async function renderBatch() {
    const pulls = await gh('/repos/' + REPO + '/pulls?state=open&per_page=50');
    const batches = pulls.filter((pr) => pr.head.ref.startsWith('editorial/'));
    if (!batches.length) { $('desk-batch').innerHTML = '<p>열린 배치 PR이 없습니다.</p>'; return; }
    const rows = await Promise.all(batches.map(async (pr) => {
      const detail = await gh('/repos/' + REPO + '/pulls/' + pr.number);
      return '<tr><td><a href="' + pr.html_url + '" target="_blank" rel="noopener">#'
        + pr.number + '</a></td><td>' + escapeHtml(pr.title) + '</td><td>'
        + detail.commits + '커밋 · ' + detail.changed_files + '파일</td><td>'
        + dateOf(pr.updated_at) + '</td></tr>';
    }));
    $('desk-batch').innerHTML =
      '<div style="overflow-x:auto"><table class="table table-sm"><thead><tr>'
      + '<th>PR</th><th>제목</th><th>규모</th><th>갱신</th></tr></thead><tbody>'
      + rows.join('') + '</tbody></table></div>';
  }

  async function renderMerged() {
    const pulls = await gh('/repos/' + REPO
      + '/pulls?state=closed&sort=updated&direction=desc&per_page=20');
    const merged = pulls.filter((pr) => pr.merged_at).slice(0, 5);
    if (!merged.length) { $('desk-merged').innerHTML = '<p>아직 병합 이력이 없습니다.</p>'; return; }
    $('desk-merged').innerHTML =
      '<ul>' + merged.map((pr) => '<li><a href="' + pr.html_url
        + '" target="_blank" rel="noopener">#' + pr.number + '</a> '
        + escapeHtml(pr.title) + ' <span class="text-muted">('
        + dateOf(pr.merged_at) + ')</span></li>').join('') + '</ul>';
  }

  function refresh() {
    [['desk-pending', renderPending], ['desk-batch', renderBatch], ['desk-merged', renderMerged]]
      .forEach(async ([id, render]) => {
        try { await render(); }
        catch (error) { $(id).innerHTML = '<p class="text-danger">' + escapeHtml(error.message) + '</p>'; }
      });
  }

  $('desk-token-form').addEventListener('submit', (event) => {
    event.preventDefault();
    const value = $('desk-token').value.trim();
    if (value) localStorage.setItem(TOKEN_KEY, value);
    $('desk-token').value = '';
    refresh();
  });
  $('desk-token-clear').addEventListener('click', () => {
    localStorage.removeItem(TOKEN_KEY);
    refresh();
  });

  refresh();
})();
</script>
```
